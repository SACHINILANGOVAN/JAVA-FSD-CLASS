1. DIAMOND PATTERN PROGRAMS:
int i,j,n=3;
for(i=0;i<=n;i+=2)
{
  for(j=n-i;j>1;j--)
  {
    System.out.print(" ");
  }
for(j=0;j<=i;j++)
{
    System.out.print("* ");
}
System.out.println();
}
for(i=n;i>0;i-=2)
{
  for(j=n-i;j<i-1;j++)
  {
    System.out.print(" ");
  }
  for(j=0;j<i-2;j++)
  {
    System.out.print("* ");
  }
  
}
